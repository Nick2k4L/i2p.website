{{- /* Render hook for ```bytefield code blocks. Parses a simple pipe-delimited format and generates styled HTML directly. No external tools needed - all styling via CSS classes. Format (one item per line): field_name | span | color | description ~ optional gap label = footer text _ | span (empty spacer to pad out a row) Colors: blue, green, yellow, purple, red, gray, orange, cyan ~ = gap (variable-length indicator), with optional label = = footer row (e.g. total length) _ = empty spacer (use to force next field onto a new row) # = comment (ignored) Blank lines are ignored. Example: expires | 4 | yellow | 4 byte date sigtype | 2 | cyan | sig type, 2 bytes _ | 2 transient_key | 8 | green | SigningPublicKey ~ */ -}}
{{- $lines := split .Inner "\n" -}} {{- range $lines -}} {{- $line := trim . " \t\r" -}} {{- if and (ne $line "") (not (strings.HasPrefix $line "#")) -}} {{- if strings.HasPrefix $line "~" -}} {{- $gapLabel := trim (strings.TrimPrefix "~" $line) " \t" -}}
{{- if ne $gapLabel "" -}} {{ $gapLabel }} {{- else -}} {{- end -}}
{{- else if strings.HasPrefix $line "=" -}} {{- $footerText := trim (strings.TrimPrefix "=" $line) " \t" -}}
{{ $footerText }}
{{- else -}} {{- $parts := split $line "|" -}} {{- $name := "" -}}{{- $span := "8" -}}{{- $color := "gray" -}}{{- $desc := "" -}} {{- if ge (len $parts) 1 -}}{{- $name = trim (index $parts 0) " \t" -}}{{- end -}} {{- if ge (len $parts) 2 -}}{{- $span = trim (index $parts 1) " \t" -}}{{- end -}} {{- if ge (len $parts) 3 -}}{{- $color = trim (index $parts 2) " \t" -}}{{- end -}} {{- if ge (len $parts) 4 -}}{{- $desc = trim (index $parts 3) " \t" -}}{{- end -}} {{- if eq $name "_" -}}
{{- else -}}
{{ $name }} {{- if ne $desc "" -}} {{ $desc }} {{- end -}}
{{- end -}} {{- end -}} {{- end -}} {{- end -}}
